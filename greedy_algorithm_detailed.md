    coins.sort(reverse=True)
    count = 0

    for coin in coins:
        count += amount // coin
        amount %= coin

def kruskal(n, edges):
    parent = list(range(n))

    def find(u):
        if parent[u] != u:
            parent[u] = find(parent[u])
        return parent[u]

    def union(u, v):
        u_root = find(u)
        v_root = find(v)
        if u_root == v_root:
            return False
        parent[v_root] = u_root
        return True

    edges.sort(key=lambda x: x[2])  # ê°€ì¤‘ì¹˜ ê¸°ì¤€ ì •ë ¬
    mst_weight = 0

    for u, v, w in edges:
        if union(u, v):
            mst_weight += w

    return mst_weight
