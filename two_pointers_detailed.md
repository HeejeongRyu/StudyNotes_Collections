## β… 4. μμ  1: λ‘ μμ ν•© (Two Sum)

```python
def two_sum(arr, target):
    arr.sort()
    left, right = 0, len(arr) - 1

    while left < right:
        total = arr[left] + arr[right]
        if total == target:
            return (arr[left], arr[right])
        elif total < target:
            left += 1
        else:
            right -= 1
    return None
```

```python
def count_subarrays_with_sum(arr, target):
    left, right = 0, 0
    total = 0
    count = 0
    n = len(arr)

    while right < n:
        total += arr[right]
        while total > target:
            total -= arr[left]
            left += 1
        if total == target:
            count += 1
        right += 1
    return count
```

```python
def max_in_sliding_window(arr, k):
    result = []
    for i in range(len(arr) - k + 1):
        result.append(max(arr[i:i+k]))
    return result
```
